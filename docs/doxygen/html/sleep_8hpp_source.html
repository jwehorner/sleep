1/**
2 * @file sleep.hpp
3 * @brief sleep.hpp defines a number of cross-platform utility functions for sleeping for
4 * millisecond and microsecond durations.
5 * @details The file provides cross platform implementations for sleeping for millisecond
6 * and microsecond durations, as well as querying the current time in microseconds.
7 * On Windows platforms the sleep functions have been optimized to provide maximum
8 * resolution in sleeping: the sleep_ms function uses waitable timers to attain
9 * slightly better resolution than sleep at little performance cost. The sleep_us
10 * function uses a combination of sleeping and busy waiting to attain the minimal
11 * error possible, at the cost of performance, thus it is advisable to use sleep_ms
12 * where possible on Windows platforms
13 * @author James Horner (James.Horner@nrc-cnrc.gc.ca or jwehorner@gmail.com)
14 * @date 2023-06-13
15 */

42 * @brief Function sleep_ms sleeps for the specified number of milliseconds.
43 * @param ms uint32_t number of milliseconds to sleep for.
44 */
45 const void sleep_ms(const uint32_t ms) {
46 struct timespec ts;
47 ts.tv_sec = ms / 1000;
48 ts.tv_nsec = ms % 1000 * 1000000;
49
50 while (nanosleep(&ts, &ts) == -1 && errno == EINTR);
51 }

132 /// Estimated minimum number of microseconds that a process can accurately sleep for using Windows timer objects.
133 const static uint64_t min_sleep_time_us = 10'000;
134 /// Portion of the remaining count that should be slept for if the minimum sleep threshold is met.

161 /// Flag for if the Windows static variables have been initialised.
162 static bool windows_timers_initialised = false;
163 /// Result of the timeBeginPeriod call to set the minimum Windows timer object resolution.
164 static MMRESULT begin_period_result;
165 /// Number of cycles per second of the Windows performance counter.
166 static uint64_t cycles_per_s;
167 /// Number of cycles per millisecond of the Windows performance counter.
168 static uint64_t cycles_per_ms;
169 /// Number of cycles per microsecond of the Windows performance counter.
170 static uint64_t cycles_per_us;
171 /// Number of cycles of the Windows performance counter that a Windows timer objects can reliably sleep for.
172 static uint64_t min_sleep_time_cycles;
173
174 /**
175 * @brief Function initialise_windows_timers initialises the static variables for using Windows
176 * timer objects for sleeping.
177 */
178 const void initialise_windows_timers() {
179 // Try to set the minimum Windows timer object resolution and store the result.
180 begin_period_result = timeBeginPeriod(1);
181
182 // Get the frequency of the Windows performance counter.
183 cycles_per_s = GetPerfFrequency();
184 // Calculate the number of cycles of the Windows performance counter per millisecond.
185 cycles_per_ms = (uint64_t) cycles_per_s / 1'000.0;
186 // Calculate the number of cycles of the Windows performance counter per microsecond.
187 cycles_per_us = (uint64_t) cycles_per_ms / 1'000.0;
188
189 // Calculate the minimum sleep time in number of cycles of the Windows performance counter.
190 min_sleep_time_cycles = min_sleep_time_us * cycles_per_us;
191
192 // Confirm that the timer variables have been initialised if there was no error in setting the timer resolution.
193 windows_timers_initialised = (begin_period_result == TIMERR_NOERROR);
194 }

220 /**
221 * @brief Function sleep_ms sleeps for the specified number of milliseconds.
222 * @param ms uint32_t number of milliseconds to sleep for.
223 */
224 const void sleep_ms(const uint32_t ms) {
225 // Initialise the Windows timer object variables if they haven't been already.
226 if (!windows_timers_initialised) initialise_windows_timers();
227
228 // Convert the number of milliseconds to 100s of nanoseconds for SetWaitableTimer.
229 LARGE_INTEGER ft;
230 ft.QuadPart = -static_cast<int64_t>(ms * 10'000);
231
232 // Create a Windows timer object to wait on.
233 HANDLE timer = CreateWaitableTimerA(NULL, TRUE, NULL);
234 // Set the timeout on the timer to the number of nanoseconds calculated earlier.
235 SetWaitableTimer(timer, &ft, 0, NULL, NULL, 0);
236 // Wait for the timer to expire.
237 WaitForSingleObject(timer, INFINITE);
238 // Close the timer once we're done with it.
239 CloseHandle(timer);
240 }

242 /**
243 * @brief Function sleep_ms_corrected sleeps for the specified number of milliseconds minus
244 * the provided schedule slip, so that the mean sleep time converges onto the desired period.
245 * @param ms uint32_t number of milliseconds to sleep for.
246 * @param error_us int64_t number of microseconds of error accumulated by sleeping.
247 * @details This function is intended to be used in the included kind of error tracking pattern,
248 * where the schedule slip is tracked with each iteration, so when the slip gets too large
249 * a sleep is skipped and the schedule catches back up.
250 * @code {.cpp}
251 * int64_t start_time_us, error_us = 0;
252 * uint32_t duration_ms = 1;
253 * while(CONDITION) {
254 * start_time_us = sleep::now_us();
255 * sleep::sleep_ms_corrected(duration_ms, error_us);
256 * error_us += ((int64_t)sleep::now_us() - start_time_us) - (duration_ms * 1'000);
257 * }
258 * @endcode
259 */
260 const void sleep_ms_corrected(const uint32_t ms, const int64_t error_us) {
261 // If the error is greater than or equal to the requested sleep duration, skip the sleep.
262 int32_t adjusted_sleep_ms = ms - (error_us / 1'000);
263 if (adjusted_sleep_ms > 0) {
264 sleep_ms(adjusted_sleep_ms);
265 }
266 return;
267 }

269 /**
270 * @brief Function sleep_us sleeps for the specified number of microseconds.
271 * @param us uint32_t number of microseconds to sleep for.
272 */
273 const void sleep_us(const uint32_t us) {
274 // Initialise the Windows timer object variables if they haven't been already.
275 if (!windows_timers_initialised) initialise_windows_timers();
276
277 // Calculate at what performance counter value we should stop sleeping.
278 uint64_t end_counter = GetPerfCounter() + us * cycles_per_us;
279
280 // If the end counter is greater than the value of the performance counter.
281 int64_t remaining_count;
282 if (end_counter > GetPerfCounter()) {
283 // Calculate how many counts are left.
284 remaining_count = end_counter - GetPerfCounter();
285 }
286 else {
287 // Else we have reached the end already.
288 return;
289 }
290
291 // While there are still cycles, we should wait.
292 while (remaining_count > 0) {
293 // If the remaining counter is greater than the sleep threshold,
294 if (remaining_count > min_sleep_time_cycles) {
295 // Try to sleep for a portion of the remaining count.
296 sleep_ms((uint32_t)((remaining_count * remaining_count_sleep_percent) / cycles_per_ms));
297 }
298 // Decrement the number of counts left.
299 remaining_count = end_counter - GetPerfCounter();
300 }
301 }
